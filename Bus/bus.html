# app.py from fastapi import FastAPI, WebSocket from fastapi.responses import HTMLResponse import asyncio import random app = FastAPI() # Dummy buses buses = [{"bus_id": f"UP{100+i}B{1000+i}", "lat": 26.0+random.random(), "lon": 80.0+random.random()} for i in range(10)] html = """ Real-time Bus Tracking

Real-time Bus Tracking Demo

""" @app.get("/") async def get(): return HTMLResponse(html) @app.websocket("/ws") async def websocket_endpoint(websocket: WebSocket): await websocket.accept() while True: # Simulate bus movement for bus in buses: bus["lat"] += random.uniform(-0.01, 0.01) bus["lon"] += random.uniform(-0.01, 0.01) await websocket.send_json(buses) await asyncio.sleep(1) # 1-second interval
